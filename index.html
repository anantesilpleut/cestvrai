// Fonction pour démarrer le timer pour un agent spécifique function startTimer(agentName) { // Afficher le champ avec le timer document.getElementById('timerContainer').style.display = 'block'; // Référence au noeud du timer const timerRef = database.ref('timer/start_time'); // Récupérer le temps de début de pause depuis la base de données Firebase timerRef.once('value', (snapshot) => { const startTime = snapshot.val(); const fifteenMinutesLater = startTime + (15 * 60 * 1000); // Mettre à jour le timer toutes les secondes const interval = setInterval(() => { const currentTime = new Date().getTime(); const remainingTime = fifteenMinutesLater - currentTime; if (remainingTime > 0) { const minutes = Math.floor((remainingTime % (1000 * 60 * 60)) / (1000 * 60)); const seconds = Math.floor((remainingTime % (1000 * 60)) / 1000); document.getElementById('timer').innerText = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`; } else { clearInterval(interval); document.getElementById('timer').innerText = '15:00'; // Réinitialiser le timer à 15:00 // Cacher le champ avec le timer document.getElementById('timerContainer').style.display = 'none'; } }, 1000); }); }
